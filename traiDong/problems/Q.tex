\section{Chọn đỉnh}\label{trai-dong-2024-q}

\subtitle{Time limit: \textbf{2 giây} \hfill Memory limit: \textbf{512}Mb}

\subsection{Đề bài}\label{trai-dong-2024-q-de-bai}

Cho đồ thị vô hướng $N$ đỉnh đánh số từ $1$ tới $N$ và có $M$
cạnh. Chọn tập đỉnh $S$ khác rỗng có tối đa $N - K$ đỉnh sao cho với
những đỉnh còn lại không được chọn thì $K$ đỉnh có chỉ số lớn nhất
không có cạnh kề với bất kì đỉnh nào trong tập $S$.

\subsection{Dữ liệu vào}\label{trai-dong-2024-q-du-lieu-vao}

\begin{itemize}
\tightlist
\item
  Dòng đầu tiên chứa $N, M, K$ ($2 \le N \leq 2\cdot 10^{5}, 1 \le M \le 4\cdot 10^5, 1 \le K \le N - 1$)
\item
  $M$ dòng tiếp theo mỗi dòng chứa hai số $u, v$ mô tả một cạnh của
  đồ thị. Dữ liệu đảm bảo không có cạnh nào nối một đỉnh với chính nó,
  không có nhiều hơn một cạnh cùng nối hai đỉnh, và đồ thị là liên thông
  hai đỉnh bất kì có thể di chuyển đến nhau.
\end{itemize}

\subsection{Dữ liệu ra}\label{trai-dong-2024-q-du-lieu-ra}

\begin{itemize}
\tightlist
\item
  Đưa ra số $C$ mô tả số lượng đỉnh chọn được, nếu không có cách chọn
  thì in ra -1.
\item
  Nếu có thể chọn được tập $S$ thì dòng thứ hai in ra chỉ số của các
  đỉnh trong tập $S$.
\end{itemize}

\subsection{Tính điểm}\label{trai-dong-2024-q-tinh-diem}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  $20\%$ số điểm có $K = 1$
\item
  $20\%$ số điểm có $N \le 15$
\item
  $20\%$ số điểm có $N \le 2000, M\le 4000$
\item
  $20\%$ số điểm có $K \le 15$
\item
  $20\%$ số điểm còn lại không có ràng buộc gì thêm.
\end{enumerate}
