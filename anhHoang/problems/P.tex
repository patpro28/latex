\section{Đường đi chi phí nhỏ nhất}\label{p.-ux111ux1b0ux1eddng-ux111i-chi-phuxed-nhux1ecf-nhux1ea5t}
\subtitle{Time limit: \textbf{1 giây} \hfill Memory limit: \textbf{256}Mb}

\subsubsection{Đề bài}\label{ux111ux1ec1-buxe0i-15}

Cho đồ thị $n$ đỉnh, các đỉnh đánh số từ $1$ đến $n$. Có $m$ cặp
đỉnh không thể di chuyển trực tiếp đến nhau. Mỗi đỉnh $u$ có một trọng
số $w_u$. Chi phí di chuyển từ đỉnh $u$ đến đỉnh $v$ là
$\lvert w_u - w_v \rvert$.

Cho phép thay đổi trọng số của không quá $k$ đỉnh $u$ mà
$1 < u < n$. Hãy tìm đường đi có chi phí nhỏ nhất từ $1$ đến $n$
khi đã thay đổi trọng số của không quá $k$ đỉnh.

\subsubsection{Dữ liệu vào}\label{dux1eef-liux1ec7u-vuxe0o-15}

\begin{itemize}
\tightlist
\item
  Dòng đầu chứa $n, m, k$
  $(2 \leq n \leq 500; 0 \leq m \leq \frac{n(n-1)}{2}; 0 \leq k \leq n-2)$.
\item
  Dòng thứ hai chứa $n$ số $w_u$ $(0 \leq w_u \leq 10^9)$.
\item
  $m$ dòng tiếp theo, mỗi dòng chứa hai số $u, v$
  $(1 \leq u \neq v \leq n)$ mô tả các cặp đỉnh không di chuyển trực
  tiếp được đến nhau.
\end{itemize}

\subsubsection{Dữ liệu ra}\label{dux1eef-liux1ec7u-ra-15}

\begin{itemize}
\tightlist
\item
  Ghi ra một số là chi phí nhỏ nhất tìm được. Nếu không thể di chuyển từ
  $1$ đến $n$ thì in ra $-1$.
\end{itemize}

\subsubsection{Scoring}\label{scoring}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  (20\%): $k = 0$
\item
  (30\%): $k = 1$
\item
  (50\%): Không có ràng buộc gì thêm.
\end{enumerate}

\subsubsection{Giải thích}\label{giux1ea3i-thuxedch-12}

\subsubsection{Ví dụ 1:}\label{vuxed-dux1ee5-1}

\begin{itemize}
\tightlist
\item
  \textbf{Không được thay đổi trọng số ($k = 0$)}:

  \begin{itemize}
  \tightlist
  \item
    Các trọng số ban đầu là $[1, 7, 4, 2]$.
  \item
    Đường đi duy nhất từ $1$ đến $4$ là $1 \to 2 \to 3 \to 4$:

    \begin{itemize}
    \tightlist
    \item
      Chi phí:
      $\lvert 1 - 7 \rvert + \lvert 7 - 4 \rvert + \lvert 4 - 2 \rvert = 6 + 3 + 2 = 11$.
    \end{itemize}
  \end{itemize}
\end{itemize}

\subsubsection{Ví dụ 2:}\label{vuxed-dux1ee5-2}

\begin{itemize}
\tightlist
\item
  \textbf{Cho phép thay đổi tối đa $k = 1$ trọng số}:

  \begin{itemize}
  \tightlist
  \item
    Thay đổi trọng số của đỉnh $2$ từ $7$ thành $1$, các trọng số
    là $[1, 1, 4, 2]$.
  \item
    Đường đi $1 \to 2 \to 3 \to 4$:

    \begin{itemize}
    \tightlist
    \item
      Chi phí:
      $\lvert 1 - 1 \rvert + \lvert 1 - 4 \rvert + \lvert 4 - 2 \rvert = 0 + 3 + 2 = 5$.
    \end{itemize}
  \item
    Đây là chi phí nhỏ nhất có thể tìm được.
  \item
    Kết quả: $5$.
  \end{itemize}
\end{itemize}
