\section{Truy vấn đầy đủ số ngắn nhất}\label{n.-truy-vux1ea5n-ux111ux1ea7y-ux111ux1ee7-sux1ed1-ngux1eafn-nhux1ea5t}
\subtitle{Time limit: \textbf{1 giây} \hfill Memory limit: \textbf{512}Mb}

\subsubsection{Đề bài}\label{ux111ux1ec1-buxe0i-13}

Cho một dãy số nguyên $a$ độ dài $n$. Bạn cần trả lời $q$ truy
vấn, truy vấn thứ $i$ đưa ra một số nguyên $x_i$, bạn cần chọn ra
một đoạn các số liên tiếp trên dãy $a$ là
$a_l, a_{l+1}, \ldots, a_r$ $(1 \leq l \leq r \leq n)$ sao cho với
mỗi số nguyên $v$ thỏa mãn $0 \leq v \leq x_i$ xuất hiện ít nhất một
lần trong đoạn đó, và độ dài của đoạn được chọn là nhỏ nhất có thể.

\subsubsection{Dữ liệu vào}\label{dux1eef-liux1ec7u-vuxe0o-13}

\begin{itemize}
\tightlist
\item
  Dòng đầu tiên chứa hai số nguyên $n, q$
  $(1 \leq n, q \leq 2 \cdot 10^5)$, tương ứng là độ dài dãy $a$ và
  số truy vấn bạn cần trả lời.
\item
  Dòng tiếp theo chứa $n$ số nguyên, số nguyên thứ $i$ là giá trị
  của $a_i$ $(0 \leq a_i \leq 10^9)$.
\item
  $q$ dòng tiếp theo, dòng thứ $i$ chứa một số nguyên $x_i$
  $(0 \leq x_i \leq 10^9)$ tương ứng là thông tin của truy vấn thứ
  $i$.
\end{itemize}

\subsubsection{Dữ liệu ra}\label{dux1eef-liux1ec7u-ra-13}

\begin{itemize}
\tightlist
\item
  Ghi ra trên $q$ dòng, dòng thứ $i$ là độ dài đoạn con liên tiếp
  ngắn nhất thỏa mãn truy vấn thứ $i$. Nếu không thể chọn ra đoạn thỏa
  mãn, ghi ra $-1$.
\end{itemize}

\subsubsection{Giới hạn}\label{giux1edbi-hux1ea1n-12}

\begin{itemize}
\tightlist
\item
  $1 \leq n, q \leq 2 \cdot 10^5$
\item
  $0 \leq a_i, x_i \leq 10^9$
\end{itemize}

\subsubsection{Subtasks}\label{subtasks-9}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  (10 điểm): $n, q, a_i, x_i \leq 200$
\item
  (10 điểm): $n \leq 200$
\item
  (10 điểm): $q = 1, x_i = 1$
\item
  (15 điểm): Các số $a_i$ đôi một phân biệt.
\item
  (15 điểm): $n \leq 2000$
\item
  (20 điểm): $q = 1$
\item
  (20 điểm): Không có ràng buộc gì thêm.
\end{enumerate}

4

\subsubsection{Giải thích}\label{giux1ea3i-thuxedch-10}

\subsubsection{\texorpdfstring{Truy vấn 1:
$x_1 = 4$}{Truy vấn 1: x\_1 = 4}}\label{truy-vux1ea5n-1-x_1-4}

\begin{itemize}
\tightlist
\item
  Để thỏa mãn truy vấn, đoạn con phải chứa tất cả các số
  $v \in \{0, 1, 2, 3, 4\}$.
\item
  Tuy nhiên, trong dãy không có phần tử $4$, do đó không thể chọn đoạn
  nào thỏa mãn.
\item
  Kết quả: $-1$.
\end{itemize}

\subsubsection{\texorpdfstring{Truy vấn 2:
$x_2 = 3$}{Truy vấn 2: x\_2 = 3}}\label{truy-vux1ea5n-2-x_2-3}

\begin{itemize}
\tightlist
\item
  Để thỏa mãn truy vấn, đoạn con phải chứa tất cả các số
  $v \in \{0, 1, 2, 3\}$.
\item
  Đoạn nhỏ nhất thỏa mãn là $[0, 1, 3, 2]$ (từ vị trí $2$ đến
  $5$), độ dài là $4$.
\item
  Kết quả: $4$.
\end{itemize}

\subsubsection{\texorpdfstring{Truy vấn 3:

\begin{itemize}
\tightlist
\item
  Để thỏa mãn truy vấn, đoạn con phải chứa tất cả các số
  $v \in \{0, 1, 2\}$.
\item
  Đoạn nhỏ nhất thỏa mãn là $[0, 1, 3, 2]$ (từ vị trí $2$ đến
  $5$), độ dài là $4$.
\item
  Kết quả: $4$.
\end{itemize}
